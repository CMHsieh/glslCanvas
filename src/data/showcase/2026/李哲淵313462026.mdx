```
// ==========================================
// ⭐️ 視覺風格與光影切換區
// ==========================================

// 【風格特效選擇器】
// 0 = 原始高斯水彩感 (平滑、半透明的筆觸)
// 1 = 色鉛筆/蠟筆畫感 (自然斑駁的紙張紋理、斜向排線、高飽和對比)
#define STYLE_MODE 1

// 【燈光特效選擇器】
// 0 = 一般燈光 (保留模型最原始的色彩)
// 1 = 動態光影 (白天 -> 黃昏 -> 晚上 -> 白天 循環過渡)
#define LIGHT_MODE 1

// ==========================================
// ⭐️ 顆粒外觀微調區 (解決太尖、太大、太突出的問題)
// ==========================================
#define PARTICLE_SIZE 0.3      // 顆粒大小 (建議 0.4 ~ 0.8)
#define PARTICLE_OPACITY 0.6   // 顆粒透明度 (建議 0.3 ~ 0.8)
#define PARTICLE_SOFTNESS 1  // 邊緣柔和度 (建議 1.2 ~ 2.0)
// ==========================================

```
